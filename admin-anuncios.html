Código HTML

Añade este snippet en cada página donde quieras mostrar banners. Carga automáticamente desde Supabase.

Copia el bloque de abajo y pégalo justo antes del </body> de cada página. Luego añade <div class="wsh-ad" data-espacio="HOME-TOP"></div> donde quieras que aparezca el banner.

1. Coloca el contenedor donde quieres el banner
<!-- Banner WSH — cambia data-espacio según la página -->
<div class="wsh-ad" data-espacio="HOME-TOP"></div>

<!-- Otros códigos posibles: -->
<!-- data-espacio="HOME-MID"        → index.html sección media -->
<!-- data-espacio="CANTERAS-TOP"    → lapidis-canteras.html    -->
<!-- data-espacio="MATERIALES-TOP"  → lapidis-materiales.html  -->
<!-- data-espacio="SECTOR-ARQ"      → sector-arquitectura.html -->
<!-- data-espacio="SECTOR-DIS"      → sector-distribuidores.html -->
<!-- data-espacio="SECTOR-FAB"      → sector-fabricantes.html  -->
<!-- data-espacio="SECTOR-CON"      → sector-constructoras.html -->
<!-- data-espacio="SECTOR-LAB"      → sector-laboratorios.html -->
<!-- data-espacio="SECTOR-LOG"      → sector-logistica.html    -->
<!-- data-espacio="SECTOR-MAQ"      → sector-maquinaria.html   -->
2. Pega este script una vez antes del </body> en cada página
<!-- WSH Ad Loader -->
<script>
(function(){
  var SB_URL = "https://whcptmdapnavcxcszwwk.supabase.co";
  var SB_KEY = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6IndoY3B0bWRhcG5hdmN4Y3N6d3drIiwicm9sZSI6ImFub24iLCJpYXQiOjE3NzMwOTk4MTAsImV4cCI6MjA4ODY3NTgxMH0.BVDpghOAPrMnt-IsVtCdgLR1OnOP_83UmeU8bfUena8";
  var today = new Date().toISOString().slice(0,10);

  // Encontrar todos los contenedores de anuncio en la página
  var slots = document.querySelectorAll('.wsh-ad[data-espacio]');
  if(!slots.length) return;

  var codigos = Array.from(slots).map(function(s){ return s.dataset.espacio; });
  var query = codigos.map(function(c){ return 'espacio_codigo.eq.' + c; }).join(',');

  fetch(SB_URL + '/rest/v1/anuncios?select=espacio_codigo,anunciante,imagen_url,url_destino,texto_alt&activo=eq.true&or=(' + query + ')&fecha_fin=gte.' + today,{
    headers:{
      'apikey': SB_KEY,
      'Authorization': 'Bearer ' + SB_KEY
    }
  })
  .then(function(r){ return r.json(); })
  .then(function(data){
    if(!data || !data.length) return;
    data.forEach(function(ad){
      slots.forEach(function(slot){
        if(slot.dataset.espacio !== ad.espacio_codigo) return;
        slot.style.cssText = 'display:block;text-align:center;padding:12px 0;';
        slot.innerHTML =
          '<a href="' + ad.url_destino + '" target="_blank" rel="noopener sponsored" ' +
          'style="display:inline-block;max-width:100%;">' +
          '<img src="' + ad.imagen_url + '" alt="' + (ad.texto_alt||ad.anunciante||'') + '" ' +
          'style="max-width:100%;height:auto;display:block;" loading="lazy"></a>';
      });
    });
  })
  .catch(function(){});
})();
</script>
